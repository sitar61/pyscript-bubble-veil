from js import document, window 
from pyodide.ffi import create_proxy 
 
# The function that generates the "veil" message 
def echo_veil(veil_level=50, veil_size=7): 
   if veil_level >= 80: 
       return f"Veil Locked: Echo Roars—{veil_size} Veils Entwine Test Light." 
   elif veil_level >= 50: 
       return f"Threshold Crests: {veil_size} Veils Yearn for Dawn's Surge." 
   else: 
       return f"Hush of Mock Awakening: {veil_size} Veils Call the Test Mane." 
 
# Setup function to attach event listeners 
def setup(): 
   slider = document.getElementById("veil_slider") 
   input_box = document.getElementById("veil_input") 
   output = document.getElementById("veil_output") 
 
   def on_input_change(event): 
       level = int(slider.value) 
       size = int(input_box.value) 
       output.innerText = echo_veil(level, size) 
 
   slider.addEventListener("input", create_proxy(on_input_change)) 
   input_box.addEventListener("input", create_proxy(on_input_change)) 
 
   # Initialize output 
   on_input_change(None) 
 
# Run setup after DOM is ready 
window.addEventListener("DOMContentLoaded", create_proxy(setup)) 
   </py-script>

Step 3: Embed in Bubble 

Open your Bubble editor. 

Drag a HTML element onto the page.

<iframe  
   src="https://your-demo-site.netlify.app/"  
   style="width:100%; height:500px; border:none;"  
   frameborder="0"  
   scrolling="no"> 
</iframe>
